#include<iostream>using namespace std;int main(){    char a[20];    cin.get(a,20);    int count=0;    bool flag=true;    for(int i=0;i<strlen(a);i++)    {        if(a[i]=='.' || (a[i]>='0' && a[i]<='9'))//排除其他字符        {            if(a[i]=='.')                count++;//标记点的个数        }        else        {            cout<<a<<" is invalid ip address"<<endl;            exit(0);        }    }    if(count==3)    {        int position = 0;        for(int i=0;i<strlen(a);)        {            int sum = 0;            while(a[position]!='.' && position<strlen(a))            {                sum *= 10;                sum += ( a[position] - '0');//转换为Int型数组                position++;                i++;            }            if(a[position]=='.')            {                position++;                i++;            }            cout<<endl<<sum<<endl;            if(sum<=0 || sum>255)            {                flag = false;                break;            }        }    }    else        flag=false;    if(flag)        cout<<a<<" is valid ip address"<<endl;    else        cout<<a<<" is invalid ip address"<<endl;    return 0;}
